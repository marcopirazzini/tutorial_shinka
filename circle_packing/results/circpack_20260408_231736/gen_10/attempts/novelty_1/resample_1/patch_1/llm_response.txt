def basin_hopping_refine(centers: np.ndarray, radii: np.ndarray, *, n_iter=40):
    """
    Global refinement using basin-hopping to escape strong local optima.

    Parameters
    ----------
    centers : np.ndarray, shape (n, 2)
    radii : np.ndarray, shape (n,)

    Returns
    -------
    centers_opt : np.ndarray, shape (n, 2)
    radii_opt : np.ndarray, shape (n,)
    """
    n = centers.shape[0]

    def objective_flat(x):
        trial_centers = x.reshape((n, 2))
        trial_centers = np.clip(trial_centers, 0.01, 0.99)
        trial_radii = compute_adaptive_radii(trial_centers)
        return -np.sum(trial_radii)

    x0 = centers.flatten()
    bounds = [(0.01, 0.99)] * (2 * n)
    minimizer_kwargs = {
        'method': 'L-BFGS-B',
        'bounds': bounds,
    }

    result = basinhopping(
        objective_flat,
        x0,
        minimizer_kwargs=minimizer_kwargs,
        niter=n_iter,
        stepsize=0.03,
        disp=False,
    )

    centers_bh = np.clip(result.x.reshape((n, 2)), 0.01, 0.99)
    radii_bh = compute_adaptive_radii(centers_bh)
    if np.sum(radii_bh) > np.sum(radii):
        return centers_bh, radii_bh

    return centers, radii
