 def initialize_concentric_rings(n):
     """
     Initialize circle centers using concentric rings.
     Larger circles near center, smaller circles at edges.
     """
     centers = np.zeros((n, 2))
     center = np.array([0.5, 0.5])
 
     # Ring configuration
     ring_config = [1, 8, 17]  # circles per ring
     radii_from_center = [0.0, 0.25, 0.42]
 
     idx = 0
     for ring_num, (count, radius) in enumerate(zip(ring_config, radii_from_center)):
         for i in range(count):
             if idx >= n:
                 break
             angle = 2.0 * np.pi * i / count
             centers[idx] = center + radius * np.array([np.cos(angle), np.sin(angle)])
             idx += 1
 
     return np.clip(centers, 0.01, 0.99)
 
 
 def initialize_greedy_maxspace(n):
     """
     Initialize by greedily placing circles to maximize available space.
     """
     centers = np.zeros((n, 2))
 
     # Start with corners and edges
     candidates = []
     for x in [0.1, 0.3, 0.5, 0.7, 0.9]:
         for y in [0.1, 0.3, 0.5, 0.7, 0.9]:
             candidates.append([x, y])
 
     np.random.seed(42)
     np.random.shuffle(candidates)
 
     for i in range(min(n, len(candidates))):
         centers[i] = candidates[i]
 
     # Fill remaining with random points
     for i in range(len(candidates), n):
         centers[i] = np.random.uniform(0.05, 0.95, 2)
 
     return np.clip(centers, 0.01, 0.99)
 
 
 def initialize_hexagonal_grid(n):
     """
     Initialize circle centers using a hexagonal grid pattern with edge bias.
     """
     centers = np.zeros((n, 2))
 
     # Generate hexagonal grid points
     hex_points = []
     rows = int(np.ceil(np.sqrt(n / 0.866)))  # Hexagonal packing density
     cols = int(np.ceil(n / rows))
 
     for row in range(rows):
         for col in range(cols):
             if len(hex_points) >= n:
                 break
             x = col * 0.5 / cols + 0.05
             y = row * 0.866 / rows + 0.05
             if row % 2 == 1:
                 x += 0.25 / cols
             hex_points.append([x, y])
 
     # Sort by distance to nearest edge (prioritize edges/corners)
     hex_points = np.array(hex_points[:n])
     edge_dist = np.minimum(
         np.minimum(hex_points[:, 0], 1.0 - hex_points[:, 0]),
         np.minimum(hex_points[:, 1], 1.0 - hex_points[:, 1])
     )
     sorted_idx = np.argsort(edge_dist)
 
     centers = np.clip(hex_points[sorted_idx], 0.01, 0.99)
     return centers
 
 
 def compute_max_radius_for_circle(idx, centers, radii):
     """
     Compute the maximum radius for circle idx given current positions and radii.
     """
     cx, cy = centers[idx]
 
     # Distance to walls
     r_wall = min(cx, cy, 1.0 - cx, 1.0 - cy)
 
     # Distance to other circles
     r_circles = float('inf')
     for j in range(len(radii)):
         if j != idx and radii[j] > 0:
             dist = np.linalg.norm(centers[idx] - centers[j])
             r_circles = min(r_circles, max(0, dist - radii[j]))
 
     return min(r_wall, r_circles)

 def compute_adaptive_radii(centers: np.ndarray) -> np.ndarray:
     """
     Adaptively compute radii using multiple aggressive sweeps and orderings.
 
     Parameters
     ----------
     centers : np.ndarray, shape (n, 2)
 
     Returns
     -------
     radii : np.ndarray, shape (n,)
     """
     n = centers.shape[0]
 
     # Start with wall-constrained radii
     radii = np.array([
         min(cx, cy, 1.0 - cx, 1.0 - cy)
         for cx, cy in centers
     ], dtype=float)
 
     # Multiple aggressive shrinking passes with different orderings
     for _ in range(15):
         # Forward pass
         for i in range(n):
             for j in range(i + 1, n):
                 dist = np.linalg.norm(centers[i] - centers[j])
                 if radii[i] + radii[j] > dist + 1e-9:
                     total = radii[i] + radii[j]
                     scale = (dist - 1e-9) / total
                     radii[i] *= scale
                     radii[j] *= scale
 
         # Backward pass with different ordering
         for i in range(n - 1, -1, -1):
             for j in range(i - 1, -1, -1):
                 dist = np.linalg.norm(centers[i] - centers[j])
                 if radii[i] + radii[j] > dist + 1e-9:
                     total = radii[i] + radii[j]
                     scale = (dist - 1e-9) / total
                     radii[i] *= scale
                     radii[j] *= scale
 
     radii = np.maximum(radii, 0.0)
     return radii
 
 
 def boost_radii(centers: np.ndarray, radii: np.ndarray) -> np.ndarray:
     """
     Post-processing to slightly increase radii by local adjustments.
 
     Parameters
     ----------
     centers : np.ndarray, shape (n, 2)
     radii : np.ndarray, shape (n,)
 
     Returns
     -------
     radii_boosted : np.ndarray, shape (n,)
     """
     n = len(radii)
     radii_boosted = radii.copy()
 
     # Try to grow each circle slightly
     for i in range(n):
         # Check wall constraint
         cx, cy = centers[i]
         max_r_wall = min(cx, cy, 1.0 - cx, 1.0 - cy)
 
         # Check overlap constraint
         max_r_overlap = max_r_wall
         for j in range(n):
             if j != i:
                 dist = np.linalg.norm(centers[i] - centers[j])
                 max_r_overlap = min(max_r_overlap, dist - radii_boosted[j] - 1e-6)
 
         radii_boosted[i] = max(radii_boosted[i], max_r_overlap)
 
     radii_boosted = np.maximum(radii_boosted, 0.0)
     return radii_boosted
 
 
 def compute_max_radii(centers: np.ndarray) -> np.ndarray:

     Parameters
     ----------
     centers : np.ndarray, shape (n, 2)
 
     Returns
     -------
     radii : np.ndarray, shape (n,)
     """
     n = centers.shape[0]
 
     # Upper bound from wall distances
     radii = np.array([
         min(cx, cy, 1.0 - cx, 1.0 - cy)
         for cx, cy in centers
     ], dtype=float)
 
     # Shrink pairwise overlapping radii proportionally (multiple passes for stability)
     for iteration in range(10):
         for i in range(n):
             for j in range(i + 1, n):
                 dist = np.linalg.norm(centers[i] - centers[j])
                 if radii[i] + radii[j] > dist + 1e-9:
                     # Proportional shrinking favors smaller circles
                     total = radii[i] + radii[j]
                     scale = (dist - 1e-9) / total
                     radii[i] *= scale
                     radii[j] *= scale
