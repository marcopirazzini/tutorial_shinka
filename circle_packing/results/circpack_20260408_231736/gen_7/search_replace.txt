<DESCRIPTION>
Introduce a lightweight simulated annealing refinement inside the main packing construction to escape local optima and boost the total sum of radii. The SA operates on both centers and radii by perturbing a small subset of centers, recomputing feasible radii, and accepting improvements (or near-improvements) with a temperature-guided probability. This complements the existing multi-start initializations and the local optimizer, aiming to push the sum of radii beyond prior bests (target > 2.635). The refinement is implemented as a nested function inside construct_packing to avoid altering the module-level API and to keep changes localized and reproducible.

def construct_packing():
    """
    Construct an arrangement of 26 non-overlapping circles inside [0,1]^2
    that attempts to maximise the sum of their radii.

    Uses multi-start initialization with adaptive radius computation, local optimization
    and a lightweight simulated annealing refinement to escape local maxima.

    Returns
    -------
    centers : np.ndarray, shape (26, 2)
        (x, y) centre coordinates, all in [0,1].
    radii : np.ndarray, shape (26,)
        Radius of each circle (non-negative, satisfying containment + non-overlap).
    """
    n = 26

    # Nested simulated annealing refinement (defined once per call)
    def _simulated_annealing_refine(centers_init, radii_init, n_steps=260, init_T=0.05, cooling=0.992):
        centers_best = centers_init.copy()
        radii_best = radii_init.copy()
        sum_best = float(np.sum(radii_best))
        T = init_T

        m = max(1, int(0.15 * n))
        for _ in range(n_steps):
            centers_candidate = centers_best.copy()
            idxs = np.random.choice(n, size=m, replace=False)
            for idx in idxs:
                centers_candidate[idx, 0] += (np.random.rand() - 0.5) * 0.15
                centers_candidate[idx, 1] += (np.random.rand() - 0.5) * 0.15
            centers_candidate = np.clip(centers_candidate, 0.01, 0.99)
            radii_candidate = compute_max_radii(centers_candidate)

            sum_candidate = float(np.sum(radii_candidate))
            delta = sum_candidate - sum_best
            if delta > 0 or np.random.rand() < np.exp(delta / max(T, 1e-9)):
                centers_best = centers_candidate
                radii_best = radii_candidate
                sum_best = sum_candidate

            T *= cooling

        return centers_best, radii_best

    best_sum = -np.inf
    best_centers = None
    best_radii = None

    init_methods = [
        lambda: initialize_concentric_rings(n),
        lambda: initialize_hexagonal_grid(n),
        lambda: initialize_greedy_maxspace(n),
    ]

    for init_method in init_methods:
        centers = init_method()

        # Adaptive radius computation with multiple sweeps
        radii = compute_adaptive_radii(centers)

        # Joint optimization of all positions and radii
        centers, radii = optimize_packing(centers, radii)

        # Post-processing: try to boost radii further
        radii = boost_radii(centers, radii)

        # Optional simulated annealing refinement to escape local optima
        centers, radii = _simulated_annealing_refine(centers, radii)

        sum_radii = float(np.sum(radii))
        if sum_radii > best_sum:
            best_sum = sum_radii
            best_centers = centers.copy()
            best_radii = radii.copy()

    return best_centers, best_radii
