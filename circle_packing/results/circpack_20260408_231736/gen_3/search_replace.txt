<DESCRIPTION>
Introduce a lightweight local-search (simulated-annealing style hill-climb) step that jointly refines circle centers to allow larger radii while maintaining non-overlap and containment. The radii are recomputed after each proposed center perturbation. This explores nearby configurations beyond the rigid-ring initialization, potentially increasing the sum of radii. The search is deterministic (fixed RNG seed) and lightweight to keep runtime reasonable, and it preserves the original interface and validity constraints.

def construct_packing():
    """
    Construct an arrangement of 26 non-overlapping circles inside [0,1]^2
    that attempts to maximise the sum of their radii.

    Returns
    -------
    centers : np.ndarray, shape (26, 2)
        (x, y) centre coordinates, all in [0,1].
    radii : np.ndarray, shape (26,)
        Radius of each circle (non-negative, satisfying containment + non-overlap).
    """
    n = 26
    centers = np.zeros((n, 2))

    # --- Simple structured initialisation ---
    # Place one circle at the centre
    centers[0] = [0.5, 0.5]

    # 8 circles in an inner ring
    for i in range(8):
        angle = 2 * np.pi * i / 8
        centers[i + 1] = [0.5 + 0.28 * np.cos(angle),
                          0.5 + 0.28 * np.sin(angle)]

    # 17 circles in an outer ring
    for i in range(17):
        angle = 2 * np.pi * i / 17
        centers[i + 9] = [0.5 + 0.44 * np.cos(angle),
                          0.5 + 0.44 * np.sin(angle)]

    # Clip centres safely inside [0, 1]
    centers = np.clip(centers, 0.01, 0.99)

    # Compute maximum valid radii given these centre positions
    radii = compute_max_radii(centers)

    # Local search: perturb centres to potentially enlarge radii
    best_centers = centers.copy()
    best_radii = radii.copy()
    best_sum = float(np.sum(best_radii))
    rng = np.random.default_rng(12345)

    # Simple simulated-annealing style hill-climb
    initial_step = 0.08
    steps = 1500
    for k in range(steps):
        # temperature-like factor decreasing over time
        t = max(0.0, 1.0 - k / steps)
        step = initial_step * (0.5 + 0.5 * t)

        # propose move: pick random circle and perturb center
        i = rng.integers(0, n)
        old_pos = best_centers[i].copy()
        delta = step * rng.normal(size=2)
        new_pos = old_pos + delta
        new_pos = np.clip(new_pos, 0.01, 0.99)

        trial_centers = best_centers.copy()
        trial_centers[i] = new_pos
        trial_radii = compute_max_radii(trial_centers)

        trial_sum = float(np.sum(trial_radii))

        if trial_sum > best_sum or rng.random() < np.exp((trial_sum - best_sum) / max(1e-6, t * 0.1)):
            best_centers[i] = new_pos
            centers = trial_centers
            radii = trial_radii
            best_centers = trial_centers
            best_radii = trial_radii
            best_sum = trial_sum

    return best_centers, best_radii
