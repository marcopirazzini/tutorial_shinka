import numpy as np

# EVOLVE-BLOCK-START
def construct_packing():
    """
    Constructs an arrangement of 26 circles in [0,1]^2 targeting a high sum of radii.
    Uses a hexagonal initialization and a deterministic pressure-based relaxation.
    """
    n = 26
    
    # 1. Hexagonal Grid Initialization
    # We fit roughly a 5x6 or 5x5 grid slightly perturbed
    cols = 5
    rows = 6
    centers = []
    for r in range(rows):
        for c in range(cols):
            if len(centers) >= n:
                break
            x = (c + 0.5 * (r % 2)) / (cols - 0.5)
            y = r / (rows - 1)
            centers.append([0.1 + 0.8 * x, 0.1 + 0.8 * y])
    
    centers = np.array(centers)
    
    # 2. Force-directed relaxation specifically designed for Sum of Radii
    # Sum of radii is maximized when circles are 'tangent' and fill gaps.
    for _ in range(150):
        centers = refine_step(centers)
    
    # Final radii computation
    radii = compute_max_radii_stable(centers)
    
    # 3. Micro-perturbation Greedy Hill Climbing
    best_centers = centers.copy()
    best_radii = radii.copy()
    best_sum = np.sum(best_radii)
    
    rng = np.random.default_rng(42)
    for _ in range(800):
        idx = rng.integers(0, n)
        old_center = best_centers[idx].copy()
        
        # Jitter
        best_centers[idx] += rng.uniform(-0.02, 0.02, size=2)
        best_centers[idx] = np.clip(best_centers[idx], 0.0, 1.0)
        
        current_radii = compute_max_radii_stable(best_centers)
        current_sum = np.sum(current_radii)
        
        if current_sum > best_sum:
            best_sum = current_sum
            best_radii = current_radii
        else:
            best_centers[idx] = old_center
            
    return best_centers, best_radii

def refine_step(centers):
    """
    Moves centers to balance pressures from walls and neighbors.
    """
    n = centers.shape[0]
    new_centers = centers.copy()
    
    for i in range(n):
        # Direction to walls
        x, y = centers[i]
        forces = np.zeros(2)
        
        # Wall repulsions (very short range)
        if x < 0.1: forces[0] += (0.1 - x)
        if x > 0.9: forces[0] -= (x - 0.9)
        if y < 0.1: forces[1] += (0.1 - y)
        if y > 0.9: forces[1] -= (y - 0.9)
        
        # Pairwise repulsion to keep spacing even
        for j in range(n):
            if i == j: continue
            diff = centers[i] - centers[j]
            dist = np.linalg.norm(diff)
            if dist < 0.18 and dist > 0:
                forces += (diff / dist) * (0.18 - dist)
        
        new_centers[i] += forces * 0.2
    
    return np.clip(new_centers, 0.0, 1.0)

def compute_max_radii_stable(centers):
    """
    Calculates radii using an equitable distribution of space between neighbors.
    """
    n = centers.shape[0]
    # Start with max possible (distance to walls)
    radii = np.array([min(c[0], 1-c[0], c[1], 1-c[1]) for c in centers])
    
    # Iterate to enforce r_i + r_j <= dist_ij
    # We solve this by allocating the gap proportionally
    for _ in range(10):
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                overlap = radii[i] + radii[j] - dist
                if overlap > 0:
                    # Reduce both based on current size ratio
                    total_r = radii[i] + radii[j]
                    radii[i] -= overlap * (radii[i] / total_r)
                    radii[j] -= overlap * (radii[j] / total_r)
    
    return np.maximum(radii, 0.0)
# EVOLVE-BLOCK-END

def run_packing():
    centers, radii = construct_packing()
    sum_of_radii = float(np.sum(radii))
    return centers, radii, sum_of_radii
