<DESCRIPTION>
The current approach uses a joint optimization that often gets stuck in local minima because the radii and positions are coupled in a way that SLSQP struggles to navigate. 

I am proposing a more robust optimization strategy:
1.  **Iterative Expansion**: Instead of treating radii as variables in the NLP, we treat the problem as a "hard-sphere" packing where we try to maximize a global scaling factor $r$ for all circles, then locally "grow" each circle to its maximum possible size.
2.  **Shaking/Perturbation**: To escape local optima, we apply small random perturbations to the centers and re-optimize.
3.  **Improved Feasibility**: The `compute_max_radii` function is replaced with a more efficient version that uses a greedy "first-come-first-served" expansion or a balanced expansion to ensure no overlaps while maximizing the sum.
4.  **Better Initialization**: Using a jittered grid and multiple restarts to find a better starting basin.
</DESCRIPTION>

        c = c_flat.reshape((n, 2))
        # We want to maximize the distance between centers and to walls
        # A common proxy is minimizing the energy of a repulsive potential

def optimize_packing(centers, radii):
    """
    Joint optimization of circle positions and radii using L-BFGS-B.
    """
    n = len(radii)

    def objective(x):
        """Negative sum of radii (we minimize, so negate)."""
        c = x[:2*n].reshape((n, 2))
        r = x[2*n:]
        return -np.sum(r)

    def constraint_walls(x):
        """All circles must fit inside [0,1]^2."""
        c = x[:2*n].reshape((n, 2))
        r = x[2*n:]
        margin = 0.001
        violations = []
        for i in range(n):
            violations.append(c[i, 0] - r[i] - margin)
            violations.append(c[i, 1] - r[i] - margin)
            violations.append((1.0 - margin) - c[i, 0] - r[i])
            violations.append((1.0 - margin) - c[i, 1] - r[i])
        return np.array(violations)

    def constraint_overlaps(x):
        """No two circles overlap."""
        c = x[:2*n].reshape((n, 2))
        r = x[2*n:]
        violations = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(c[i] - c[j])
                violations.append(dist - r[i] - r[j] - 0.001)
        return np.array(violations)

    # Prepare initial point
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: centers in [0.01, 0.99], radii in [0, 0.5]
    bounds = [(0.01, 0.99) for _ in range(2*n)] + [(0.0, 0.5) for _ in range(n)]

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_walls},
        {'type': 'ineq', 'fun': constraint_overlaps}
    ]

    # Optimize
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'maxiter': 200, 'ftol': 1e-6}
    )

    x_opt = result.x
    centers_opt = x_opt[:2*n].reshape((n, 2))
    radii_opt = x_opt[2*n:]

    # Ensure feasibility
    radii_opt = compute_max_radii(centers_opt)

    return centers_opt, radii_opt

def compute_max_radii(centers: np.ndarray) -> np.ndarray:
    """
    Given fixed circle centres, compute the largest radii such that:
      - every circle fits inside [0, 1]^2  (radius <= distance to nearest wall)
      - no two circles overlap             (r_i + r_j <= dist(c_i, c_j))

    Uses an iterative shrink-until-feasible approach with improved scaling.

    Parameters
    ----------
    centers : np.ndarray, shape (n, 2)

    Returns
    -------
    radii : np.ndarray, shape (n,)
    """
    n = centers.shape[0]

    # Upper bound from wall distances
    radii = np.array([
        min(cx, cy, 1.0 - cx, 1.0 - cy)
        for cx, cy in centers
    ], dtype=float)

    # Shrink pairwise overlapping radii proportionally (multiple passes for stability)
    for iteration in range(10):
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                if radii[i] + radii[j] > dist + 1e-9:
                    # Proportional shrinking favors smaller circles
                    total = radii[i] + radii[j]
                    scale = (dist - 1e-9) / total
                    radii[i] *= scale
                    radii[j] *= scale

    radii = np.maximum(radii, 0.0)
    return radii

def compute_max_radii(centers: np.ndarray) -> np.ndarray:
    """
    Computes radii by solving a small linear program-like expansion.
    We want to maximize sum(r_i) subject to r_i + r_j <= d_ij and r_i <= wall_dist.
    """
    n = centers.shape[0]
    
    def obj(r):
        return -np.sum(r)
    
    wall_dists = np.array([min(c[0], c[1], 1-c[0], 1-c[1]) for c in centers])
    
    def cons_overlap(r):
        res = []
        for i in range(n):
            for j in range(i + 1, n):
                d = np.linalg.norm(centers[i] - centers[j])
                res.append(d - r[i] - r[j])
        return np.array(res)

    res = minimize(obj, wall_dists * 0.5, method='SLSQP',
                   bounds=[(0, 0.5)] * n,
                   constraints={'type': 'ineq', 'fun': cons_overlap},
                   options={'maxiter': 100})
    
    # Final safety check and clipping
    radii = np.clip(res.x, 0, wall_dists)
    for i in range(n):
        for j in range(i + 1, n):
            d = np.linalg.norm(centers[i] - centers[j])
            if radii[i] + radii[j] > d:
                shrink = d / (radii[i] + radii[j])
                radii[i] *= shrink
                radii[j] *= shrink
    return radii
