<DESCRIPTION>
Introduce a lightweight physics-inspired relaxation step to spread circles more efficiently before the local search. This helps avoid cramped initial configurations and allows larger radii to emerge from the compute_max_radii step. The improvement comprises:
- A new relax_layout that gently repels overlapping circles and nudges them away from walls.
- A call to relax_layout immediately after the initial placement, prior to the simulated annealing local search.
- A more aggressive local search (larger initial step and more iterations) to leverage the better starting arrangement.

Overall, this should yield larger total radii and move towards the better-known benchmarks (closer to 2.6+).

def compute_max_radii(centers: np.ndarray) -> np.ndarray:
    """
    Given fixed circle centres, compute the largest radii such that:
      - every circle fits inside [0, 1]^2  (radius <= distance to nearest wall)
      - no two circles overlap             (r_i + r_j <= dist(c_i, c_j))

    Uses an iterative shrink-until-feasible approach.

    Parameters
    ----------
    centers : np.ndarray, shape (n, 2)

    Returns
    -------
    radii : np.ndarray, shape (n,)
    """
    n = centers.shape[0]

    # Upper bound from wall distances
    radii = np.array([
        min(cx, cy, 1.0 - cx, 1.0 - cy)
        for cx, cy in centers
    ], dtype=float)

    # Shrink pairwise overlapping radii proportionally (multiple passes for stability)
    for _ in range(5):
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                if radii[i] + radii[j] > dist:
                    scale = dist / (radii[i] + radii[j])
                    radii[i] *= scale
                    radii[j] *= scale

    radii = np.maximum(radii, 0.0)
    return radii

def relax_layout(centers: np.ndarray, iterations: int = 200, step: float = 0.08) -> np.ndarray:
    """
    Relaxes the layout of circle centers using simple repulsive adjustments.
    Returns a new array of centers after a few relaxation passes.
    Radii are not modified here; they are recomputed as needed by the caller.
    """
    centers = centers.copy()
    rng = np.random.default_rng(123456)
    for _ in range(iterations):
        radii = compute_max_radii(centers)
        n = centers.shape[0]
        disp = np.zeros_like(centers)
        # Pairwise repulsion
        for i in range(n):
            for j in range(i + 1, n):
                diff = centers[i] - centers[j]
                dist = np.hypot(diff[0], diff[1])
                if dist < 1e-12:
                    diff = rng.normal(size=2) * 1e-3
                    dist = np.hypot(diff[0], diff[1])
                overlap = radii[i] + radii[j] - dist
                if overlap > 0:
                    dir = diff / dist
                    push = min(overlap * 0.6, step)
                    disp[i] += dir * push
                    disp[j] -= dir * push
        # Gentle wall repulsion
        for i in range(n):
            cx, cy = centers[i]
            if cx < 0.25:
                disp[i,0] += 0.02
            elif cx > 0.75:
                disp[i,0] -= 0.02
            if cy < 0.25:
                disp[i,1] += 0.02
            elif cy > 0.75:
                disp[i,1] -= 0.02
        centers += disp
        centers = np.clip(centers, 0.01, 0.99)
    return centers
