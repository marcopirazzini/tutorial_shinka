    """
    Construct an arrangement of 26 non-overlapping circles inside [0,1]^2
    that attempts to maximise the sum of their radii.

    Returns
    -------
    centers : np.ndarray, shape (26, 2)
        (x, y) centre coordinates, all in [0,1].

    """

        # Inverse: smaller radii get higher probability

        old_pos = best_centers[i].copy()
        delta = step * rng.normal(size=2)
        new_pos = old_pos + delta
        new_pos = np.clip(new_pos, 0.01, 0.99)

        trial_centers = best_centers.copy()
        trial_centers[i] = new_pos
        trial_radii = compute_max_radii(trial_centers)
        trial_sum = float(np.sum(trial_radii))

        accept = trial_sum > best_sum
        if not accept and t > 1e-6:
            accept = rng.random() < np.exp((trial_sum - best_sum) / max(1e-6, t * 0.15))

        if accept:
            best_centers = trial_centers
            best_radii = trial_radii
            best_sum = trial_sum

def _generate_candidate_positions(grid_res: int = 8) -> np.ndarray:
    """Generate candidate positions biased toward corners and edges."""
    candidates = []

    # Corner regions (high priority - walls provide constraint)
    corner_margin = 0.15
    for cx in [corner_margin, 0.5, 1.0 - corner_margin]:
        for cy in [corner_margin, 0.5, 1.0 - corner_margin]:
            if not (0.4 < cx < 0.6 and 0.4 < cy < 0.6):  # Avoid center cluster
                candidates.append([cx, cy])

    # Edge positions
    for x in np.linspace(0.1, 0.9, 6):
        candidates.append([x, 0.1])
        candidates.append([x, 0.9])
        candidates.append([0.1, x])
        candidates.append([0.9, x])

    # Interior grid
    for x in np.linspace(0.2, 0.8, grid_res):
        for y in np.linspace(0.2, 0.8, grid_res):
            candidates.append([x, y])

    return np.array(candidates, dtype=float)


def _greedy_place_circles(n: int, candidates: np.ndarray, rng: np.random.Generator) -> np.ndarray:
    """Greedily place n circles from candidate positions."""
    centers = []

    for circle_idx in range(n):
        best_score = -np.inf
        best_pos = None

        # Evaluate each candidate
        for cand in candidates:
            if np.any(np.abs(cand - np.array(centers)) < 1e-6 if len(centers) > 0 else False):
                continue  # Skip already used positions

            # Score based on distance to walls and existing circles
            wall_dist = min(cand[0], cand[1], 1.0 - cand[0], 1.0 - cand[1])

            if len(centers) == 0:
                score = wall_dist
            else:
                centers_arr = np.array(centers)
                min_dist_to_existing = np.min(np.linalg.norm(centers_arr - cand, axis=1))
                # Prefer positions far from walls (interior) for large circles
                # and close to walls for small circles (will be optimized later)
                score = wall_dist + 0.5 * min_dist_to_existing

            if score > best_score:
                best_score = score
                best_pos = cand.copy()

        if best_pos is not None:
            centers.append(best_pos)
        else:
            # Fallback: random position
            centers.append(rng.uniform(0.01, 0.99, size=2))

    return np.array(centers, dtype=float)

def relax_layout(centers: np.ndarray, iterations: int = 200, step: float = 0.08) -> np.ndarray:

    Returns a new array of centers after a few relaxation passes.
    Radii are not modified here; they are recomputed as needed by the caller.
    """

    rng = np.random.default_rng(123456)

        radii = compute_max_radii(centers)
        n = centers.shape[0]

        # Pairwise repulsion
        for i in range(n):
            for j in range(i + 1, n):
                diff = centers[i] - centers[j]
                dist = np.hypot(diff[0], diff[1])

def relax_layout_radius_aware(centers: np.ndarray, iterations: int = 250, step: float = 0.10) -> np.ndarray:
    """
    Relaxes the layout of circle centers using repulsive adjustments weighted by radius.
    Larger circles contribute more to repulsion forces.
    """
    centers = centers.copy()
    rng = np.random.default_rng(123456)

    for iteration in range(iterations):
        radii = compute_max_radii(centers)
        n = centers.shape[0]
        disp = np.zeros_like(centers)

        # Normalize radii for weighting
        max_radius = np.max(radii) + 1e-9
        radius_weights = radii / max_radius

        # Pairwise repulsion with radius weighting
        for i in range(n):
            for j in range(i + 1, n):
                diff = centers[i] - centers[j]
                dist = np.hypot(diff[0], diff[1])
                if dist < 1e-12:
                    diff = rng.normal(size=2) * 1e-3
                    dist = np.hypot(diff[0], diff[1])

                overlap = radii[i] + radii[j] - dist
                if overlap > 0:
                    direction = diff / dist
                    # Weight by radius: larger circles push harder
                    weight_i = radius_weights[i]
                    weight_j = radius_weights[j]
                    total_weight = weight_i + weight_j + 1e-9

                    push = min(overlap * 0.7, step)
                    disp[i] += direction * push * (weight_j / total_weight)
                    disp[j] -= direction * push * (weight_i / total_weight)

        # Boundary repulsion - gentle but consistent
        for i in range(n):
            cx, cy = centers[i]
            margin = 0.15

            if cx < margin:
                disp[i, 0] += (margin - cx) * 0.1
            elif cx > 1.0 - margin:
                disp[i, 0] -= (cx - (1.0 - margin)) * 0.1

            if cy < margin:
                disp[i, 1] += (margin - cy) * 0.1
            elif cy > 1.0 - margin:
                disp[i, 1] -= (cy - (1.0 - margin)) * 0.1

        centers += disp
        centers = np.clip(centers, 0.01, 0.99)

        # Cooling: reduce step size over iterations
        step *= (1.0 - iteration / (iterations * 2.0))

    return centers
