<DESCRIPTION>
Introduce a more robust global search strategy: switch from a single multi-start local search to a diversified, seed-driven simulated annealing (SA) over centers, with LP-derived radii as the evaluation. This targets escaping local optima and exploring denser configurations (corner/edge placements often being critical). Key components:
- A redesigned seed generator that builds dense hex/near-hex arrangements and fills gaps deterministically, then adds small jitter.
- An inner SA routine that perturbs centers, re-evaluates radii via compute_optimal_radii (LP), and accepts moves using a Metropolis-like criterion with a cooling schedule.
- Multiple seeds to improve global coverage, plus an optional final refinement step using LBFGS-B to polish the best found centers.
- Keeps the required interface and data shapes; higher sum of radii should improve the score and potentially beat the 2.635 target.

This change is self-contained and leverages the existing LP-based radii computation, but uses a more principled stochastic/global search to push beyond previous bests.

def construct_packing():
    """
    Uses a diversified seed-driven simulated annealing (SA) with LP-based radii evaluation.
    Aims to explore the center-space more broadly than a single multi-start LBFGS-B.
    """
    n = 26

    def compute_radii_lp(centers_np):
        return compute_optimal_radii(np.asarray(centers_np))

    def objective(flat_centers):
        centers_now = flat_centers.reshape((n, 2))
        r = compute_radii_lp(centers_now)
        return -float(np.sum(r))

    def generate_seed_centers(seed_id):
        rng = np.random.default_rng(seed_id)
        centers = np.zeros((n, 2))
        centers[0] = [0.5, 0.5]

        # Dense hex-like lattice seed
        positions = []
        dx = 0.10
        dy = dx * np.sqrt(3) / 2
        for row in range(-2, 3):
            y = 0.5 + row * dy
            if not (0.05 <= y <= 0.95):
                continue
            shift = 0.0 if (row % 2 == 0) else dx / 2.0
            for col in range(-3, 4):
                x = 0.5 + col * dx + shift
                if 0.05 <= x <= 0.95 and not (abs(x - 0.5) < 1e-9 and abs(y - 0.5) < 1e-9):
                    positions.append((x, y))
                    if len(positions) >= n - 1:
                        break
            if len(positions) >= n - 1:
                break

        for i, p in enumerate(positions, start=1):
            centers[i] = p

        if len(positions) < n - 1:
            # Fill remainder with jittered random points in the square
            for i in range(1 + len(positions), n):
                centers[i] = [0.5 + rng.uniform(-0.45, 0.45),
                              0.5 + rng.uniform(-0.45, 0.45)]
        centers = np.clip(centers, 0.05, 0.95)
        centers += rng.normal(scale=0.012, size=centers.shape)
        centers = np.clip(centers, 0.05, 0.95)
        return centers

    def anneal(seed_centers, rng, steps=700, init_scale=0.04):
        current_centers = seed_centers.copy()
        current_radii = compute_optimal_radii(current_centers)
        current_score = float(np.sum(current_radii))
        best_centers = current_centers.copy()
        best_radii = current_radii.copy()
        best_score = current_score

        for step in range(steps):
            t = (steps - step) / steps
            scale = init_scale * (t ** 0.5)
            cand = current_centers + rng.normal(scale=scale, size=current_centers.shape)
            cand = np.clip(cand, 0.0, 1.0)
            cand_radii = compute_optimal_radii(cand)
            cand_score = float(np.sum(cand_radii))

            delta = cand_score - current_score
            temp = 0.05 * t + 1e-9
            if delta > 0 or rng.random() < np.exp(delta / temp):
                current_centers = cand
                current_radii = cand_radii
                current_score = cand_score

                if cand_score > best_score:
                    best_score = cand_score
                    best_centers = cand
                    best_radii = cand_radii

        return best_centers, best_radii, best_score

    seeds = [111, 222, 333, 444, 555, 666]
    best_centers = None
    best_radii = None
    best_score = -np.inf

    for sid in seeds:
        seed_centers = generate_seed_centers(sid)
        rng = np.random.default_rng(sid)
        centers_candidate, radii_candidate, score_candidate = anneal(seed_centers, rng, steps=700, init_scale=0.04)

        # Optional final polish via LBFGS-B on the best candidate
        try:
            res = minimize(
                lambda flat: -float(np.sum(compute_optimal_radii(flat.reshape((n, 2))))) ,
                centers_candidate.flatten(),
                method='L-BFGS-B',
                bounds=[(0, 1)] * (2 * n),
                options={'maxiter': 250, 'ftol': 1e-6, 'gtol': 1e-6}
            )
            if res.success:
                centers_candidate = res.x.reshape((n, 2))
                radii_candidate = compute_optimal_radii(centers_candidate)
                score_candidate = float(np.sum(radii_candidate))
        except Exception:
            pass  # fall back to SA result if LBFGS-B fails

        if score_candidate > best_score:
            best_score = score_candidate
            best_centers = centers_candidate
            best_radii = radii_candidate

    # Final safeguard: ensure radii computed from best centers
    if best_centers is None:
        best_centers = np.zeros((n, 2))
        best_centers[0] = [0.5, 0.5]
        best_radii = compute_optimal_radii(best_centers)

    best_radii = compute_optimal_radii(best_centers)
    return best_centers, best_radii
