<DESCRIPTION>
Introduce a joint center-and-radius optimization (penalty-based NLP) to further push the sum of radii higher than the previous LP-based approach. The idea is to optimize both centers and radii simultaneously, with a smooth penalty for any wall/overlap violations, allowing the optimizer to explore configurations that LP-decoupled radii may miss. This should help approach the known 2.635 bound for 26 circles. The change replaces the previous multi-start annealing+LP pipeline with a single, more aggressive joint optimization loop (still seeded from the existing layout) that uses L-BFGS-B on an augmented variable vector [centers (52) , radii (26)]. The objective maximizes sum(r_i) while penalizing wall distances and pairwise overlaps via a quadratic penalty, controlled by a large penalty weight. Multiple small perturbation restarts are performed to improve robustness, and the final radii are taken from the best run. This preserves the required interface and shapes, while providing a principled, unified optimization that often yields a larger total radius.

def construct_packing():
    """
    Joint optimization: optimize centers and radii together with penalties
    to maximize the sum of radii while respecting wall and pairwise constraints.
    Uses a penalty-based NLP solved by L-BFGS-B with multiple restarts.
    """
    n = 26
    base_centers = np.array([
        [0.5107, 0.5061],
        [0.7840, 0.5052],
        [0.7146, 0.7022],
        [0.5194, 0.7810],
        [0.3082, 0.7176],
        [0.2332, 0.5077],
        [0.2962, 0.2926],
        [0.5110, 0.2276],
        [0.7129, 0.3045],
        [0.9419, 0.5054],
        [0.9086, 0.6512],
        [0.8836, 0.8576],
        [0.6762, 0.9039],
        [0.5258, 0.9411],
        [0.3817, 0.9116],
        [0.1965, 0.9029],
        [0.0951, 0.7396],
        [0.0714, 0.5747],
        [0.0738, 0.4295],
        [0.0887, 0.2675],
        [0.1825, 0.1018],
        [0.3740, 0.0901],
        [0.5230, 0.0618],
        [0.6781, 0.0977],
        [0.8841, 0.1534],
        [0.9087, 0.3596],
    ], dtype=float)

    # Objective: maximize sum(r_i) with penalties for wall/overlap
    def _joint_objective(flat):
        centers = flat[: 2 * n].reshape((n, 2))
        radii = flat[2 * n:]
        radii = radii.astype(float)

        # wall-limit for each circle
        limits = np.array([min(cx, cy, 1.0 - cx, 1.0 - cy) for cx, cy in centers])
        wall_penalty = np.maximum(radii - limits, 0.0) ** 2

        # pairwise overlap penalties
        pair_penalty = 0.0
        for i in range(n):
            for j in range(i + 1, n):
                d = np.linalg.norm(centers[i] - centers[j])
                pair_penalty += np.maximum(radii[i] + radii[j] - d, 0.0) ** 2

        penalty = wall_penalty.sum() + pair_penalty
        # weight large enough to enforce constraints but still allow exploration
        lam = 1000.0
        return -np.sum(radii) + lam * penalty

    # Bounds: centers in [0,1], radii in [0, 0.6] (safe upper bound)
    bounds = [(0.0, 1.0)] * (2 * n) + [(0.0, 0.6)] * n

    # Initial seed: use base layout radii from LP-based solution
    initial_centers = base_centers.copy()
    initial_radii = compute_optimal_radii(initial_centers)
    x0 = np.concatenate([initial_centers.flatten(), initial_radii])

    # Helper to run a single optimization
    def run_once(seed, centers_seed):
        rng = np.random.default_rng(seed)
        # perturb centers a bit for diversity
        centers = np.clip(centers_seed + rng.normal(scale=0.008, size=centers_seed.shape), 0.0, 1.0)
        radii = compute_optimal_radii(centers)
        radii = np.clip(radii, 0.0, 0.6)
        x = np.concatenate([centers.flatten(), radii])
        res = minimize(_joint_objective, x, method="L-BFGS-B", bounds=bounds,
                       options={"maxiter": 400, "ftol": 1e-7, "gtol": 1e-7})
        if res.success and res.x is not None:
            cen = res.x[: 2 * n].reshape((n, 2))
            rad = res.x[2 * n:]
            rad = np.maximum(rad, 0.0)
            score = float(np.sum(rad))
            return cen, rad, score
        return centers, radii, float(np.sum(radii))

    # Run a few restarts with different seeds
    restarts = [
        (11, initial_centers),
        (37, initial_centers),
        (73, initial_centers),
        (157, initial_centers)
    ]
    best_centers = initial_centers
    best_radii = initial_radii
    best_score = float(np.sum(best_radii))

    for seed, seed_centers in restarts:
        cen, rad, score = run_once(seed, seed_centers)
        if score > best_score:
            best_centers, best_radii, best_score = cen, rad, score

    # Final polish: small gradient step to fine-tune
    def _final_objective(flat):
        return _joint_objective(flat)

    flat0 = np.concatenate([best_centers.flatten(), best_radii])
    res = minimize(_final_objective, flat0, method="L-BFGS-B", bounds=bounds,
                   options={"maxiter": 200, "ftol": 1e-7, "gtol": 1e-7})
    if res.success and res.x is not None:
        final_centers = res.x[: 2 * n].reshape((n, 2))
        final_radii = res.x[2 * n:]
        final_radii = np.maximum(final_radii, 0.0)
    else:
        final_centers = best_centers
        final_radii = best_radii

    return final_centers, final_radii
