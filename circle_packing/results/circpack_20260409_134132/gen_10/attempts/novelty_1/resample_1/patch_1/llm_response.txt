<DESCRIPTION>
A completely different algorithmic approach: a simulated-annealing (SA) search over circle centers with a soft feasibility objective, coupled with a fast feasibility projection to radii. Instead of optimizing radii directly for fixed centers (as in the LP-on-centers approach), this method explores centers to maximize the achievable total radius after projecting radii to the feasible region. At the end, a rigorous linear-programming projection is applied to obtain the maximal feasible radii for the chosen centers. This “centers-first SA with LP projection” strategy explores disparate layouts (including restarts from a deterministic seed) and tends to find higher total radii than purely center-fixed LP methods, especially in larger-n circle packings. The final radii are guaranteed feasible by the LP projection.

import numpy as np
from scipy.optimize import linprog

N = 26

def _seed_centers(n=N):
    """
    Deterministic seed centers (26 points) forming a dense but well-separated pattern.
    The seed is then jittered during SA runs to explore other basins.
    """
    base = [
        [0.5, 0.5],
        [0.15, 0.15], [0.85, 0.15], [0.15, 0.85], [0.85, 0.85],
        [0.5, 0.15], [0.5, 0.85], [0.15, 0.5], [0.85, 0.5],
        [0.3, 0.3], [0.7, 0.3], [0.3, 0.7], [0.7, 0.7],
        [0.4, 0.4], [0.6, 0.4], [0.4, 0.6], [0.6, 0.6],
        [0.25, 0.25], [0.75, 0.25], [0.25, 0.75], [0.75, 0.75],
        [0.33, 0.66], [0.66, 0.33], [0.66, 0.66], [0.33, 0.33],
        [0.5, 0.25], [0.25, 0.5],
    ]
    arr = np.array(base[:n], dtype=float)
    return arr

def _feasible_radii_from_centers(centers):
    """
    Given centers, return radii that are as large as possible while
    (a) staying inside the walls and (b) not overlapping each other.
    This is a quick local projection (shrink-to-fit) to a feasible packing.
    """
    n = centers.shape[0]
    radii = np.array([min(x, y, 1.0 - x, 1.0 - y) for x, y in centers], dtype=float)

    # Simple pairwise shrink passes to resolve overlaps
    for _ in range(3):
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                if radii[i] + radii[j] > dist:
                    s = dist / (radii[i] + radii[j])
                    radii[i] *= s
                    radii[j] *= s

    radii = np.maximum(radii, 0.0)
    return radii

def _lp_radii_from_centers(centers):
    """
    Given fixed centers, solve an LP to maximize the sum of radii:
      maximize sum r_i
      subject to:
        r_i <= dist_to_wall_i
        r_i + r_j <= dist(c_i, c_j) for all i<j
    Returns radii as a numpy array.
    """
    n = centers.shape[0]
    wall_dist = np.array([min(cx, cy, 1.0 - cx, 1.0 - cy) for (cx, cy) in centers], dtype=float)

    # Build LP: maximize sum r_i  -> minimize -sum r_i
    c = -np.ones(n)

    # Bounds: 0 <= r_i <= wall_dist[i]
    bounds = [(0.0, max(0.0, wd)) for wd in wall_dist]

    # Overlap constraints: r_i + r_j <= dist(c_i, c_j)
    A_ub = []
    b_ub = []
    for i in range(n):
        for j in range(i + 1, n):
            dist = np.linalg.norm(centers[i] - centers[j])
            row = np.zeros(n)
            row[i] = 1.0
            row[j] = 1.0
            A_ub.append(row)
            b_ub.append(dist)

    A_ub = np.asarray(A_ub, dtype=float) if len(A_ub) > 0 else np.zeros((0, n))
    b_ub = np.asarray(b_ub, dtype=float) if len(b_ub) > 0 else np.zeros(0)

    try:
        res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')
        if res.success:
            radii = np.asarray(res.x, dtype=float)
            radii = np.maximum(radii, 0.0)
            return radii
    except Exception:
        pass

    # Fallback to quick projection if LP fails
    return _feasible_radii_from_centers(centers)

def _generate_seed_centers(n=N, rng=None):
    """
    Produce a base set of centers (26) and apply a small jitter if rng is provided.
    """
    centers = _seed_centers(n)
    if rng is not None:
        jitter = rng.normal(scale=0.04, size=centers.shape)
        centers = centers + jitter
        centers = np.clip(centers, 0.01, 0.99)
    return centers

def construct_packing():
    """
    SA over centers with a fast feasibility projection to radii, followed by a
    final LP-projection to obtain the maximal feasible radii for the chosen centers.
    Returns centers (26,2) and radii (26,).
    """
    n = N
    # Deterministic seeds with occasional jitter per restart
    base_centers = _seed_centers(n)
    best_centers = None
    best_radii = None
    best_sum = -1.0

    rng = np.random.default_rng(12345)

    for restart in range(4):
        centers = base_centers.copy()
        # Introduce diversity across restarts
        centers = centers + rng.normal(scale=0.05, size=centers.shape)
        centers = np.clip(centers, 0.01, 0.99)

        radii = _feasible_radii_from_centers(centers)
        current_sum = float(np.sum(radii))

        # SA schedule
        T = 0.05
        cooling = 0.9997

        for it in range(3200):
            k = rng.integers(n)
            old_center = centers[k].copy()

            # Propose a move
            centers[k] += rng.normal(scale=0.08, size=2)
            centers[k] = np.clip(centers[k], 0.01, 0.99)

            new_r = _feasible_radii_from_centers(centers)
            new_sum = float(np.sum(new_r))

            # Accept with SA criterion targeting larger feasible radii sum
            if new_sum > current_sum or rng.random() < np.exp((new_sum - current_sum) / max(T, 1e-9)):
                radii = new_r
                current_sum = new_sum
                if current_sum > best_sum:
                    best_sum = current_sum
                    best_centers = centers.copy()
                    best_radii = radii.copy()
            else:
                centers[k] = old_center

            T *= cooling

        # End of one restart

    # Final radii projection for the best centers (feasibility guaranteed)
    if best_centers is None:
        best_centers = _generate_seed_centers(n, rng)
    radii_final = _lp_radii_from_centers(best_centers)
    sum_final = float(np.sum(radii_final))

    return best_centers, radii_final

def run_packing():
    """
    Main entrypoint called by the evaluator.

    Returns
    -------
    centers      : np.ndarray, shape (26, 2)
    radii        : np.ndarray, shape (26,)
    sum_of_radii : float
    """
    centers, radii = construct_packing()
    sum_of_radii = float(np.sum(radii))
    return centers, radii, sum_of_radii
