def construct_packing():
    """
    Constructs a 26-circle packing using basinhopping over centers
    with LP-based radii optimization to maximize the sum of radii.
    """
    n = 26
    # Deterministic symmetric initial layout: center, 8 around, 17 around
    centers_init = np.zeros((n, 2))
    centers_init[0] = [0.5, 0.5]
    for i in range(8):
        angle = 2 * np.pi * i / 8
        centers_init[i + 1] = [0.5 + 0.28 * np.cos(angle),
                               0.5 + 0.28 * np.sin(angle)]
    for i in range(17):
        angle = 2 * np.pi * i / 17
        centers_init[i + 9] = [0.5 + 0.44 * np.cos(angle),
                               0.5 + 0.44 * np.sin(angle)]
    centers_init = np.clip(centers_init, 0.01, 0.99)
    
    def objective_centers(flat_centers):
        centers = flat_centers.reshape(n, 2)
        radii = _compute_lp_radii(centers)
        return -float(np.sum(radii))

    bounds = [(0.0, 1.0)] * (2 * n)
    x0 = centers_init.flatten()

    from scipy.optimize import basinhopping, minimize  # local imports for reliability
    minimizer_kwargs = {"method": "L-BFGS-B", "bounds": bounds}
    result = basinhopping(objective_centers, x0, minimizer_kwargs=minimizer_kwargs,
                          niter=200, stepsize=0.2)

    centers_opt = result.x.reshape(n, 2)
    radii_opt = _compute_lp_radii(centers_opt)

    return centers_opt, radii_opt
